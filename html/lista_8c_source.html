190
191 printf("Quantidade para remover: ");
192 scanf("%d", &quantidade);

195 printf("Quantidade insuficiente no estoque.\n");
196 return;
197 }
198
199 Produto vendido = no->produto;
200 vendido.quantidade = quantidade; // Salva apenas a quantidade vendida
201
202 lista_inserir_fim(vendas, vendido); // Adiciona a venda à lista de vendas
203 fila_inserir(historicoVendas, vendido); // Adiciona a venda à fila de histórico
204
205 no->produto.quantidade -= quantidade;
206
207 if (no->produto.quantidade == 0) { // Se zerar o estoque, adiciona à reposição e remove do estoque
208 lista_inserir_fim(reposicao, no->produto);
209 lista_remover_no(estoque, no);
210 }
211
212 printf("%d unidades de %s enviadas para entrega e registradas como venda.\n", quantidade, vendido.nome);
213}
